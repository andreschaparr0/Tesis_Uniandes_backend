\chapter{Prompts utilizados}


\section*{Estructuración de CVs (\texttt{cv\_simple\_extractor.py})}
\begin{table}[H]
  \centering
  \caption{Prompts de estructuración de CV}
  \label{tab:prompts-cv}
  \footnotesize
  \begin{tabular}{@{}lp{11cm}@{}}
    \toprule
    \textbf{Parámetro} & \textbf{Prompt (resumen truncado)} \\
    \midrule
    Personal & Extrae únicamente la información personal. Responde JSON exacto: \texttt{\{"name","email","phone","location"\}}. Si falta, usar \texttt{"DESCONOCIDO"}. \\
    Educación & Extrae únicamente la información educativa. Responde JSON array de objetos: \texttt{\{"degree","institution","year","field"\}}. \\
    Experiencia & Extrae únicamente la experiencia laboral. Responde JSON array: \texttt{\{"position","company","duration","description"\}}. \\
    Habilidades técnicas & Extrae únicamente las habilidades técnicas. Responde array \texttt{["hab1","hab2",...]}. \\
    Habilidades blandas & Extrae únicamente las habilidades blandas. Responde array \texttt{["hab1","hab2",...]}. \\
    Certificaciones & Extrae únicamente las certificaciones. Responde JSON array: \texttt{\{"name","issuer","year"\}}. \\
    Idiomas & Extrae únicamente los idiomas. Responde JSON objeto \texttt{\{"idioma":"nivel",...\}}. \\
    \bottomrule
  \end{tabular}
\end{table}


\section*{Estructuración de Jobs (\texttt{job\_description\_extractor.py})}
\begin{table}[H]
  \centering
  \caption{Prompts de estructuración de descripciones de trabajo}
  \label{tab:prompts-job}
  \footnotesize
  \begin{tabular}{@{}lp{11cm}@{}}
    \toprule
    \textbf{Parámetro} & \textbf{Prompt (resumen truncado)} \\
    \midrule
    Información básica & Extrae únicamente la información básica. Responde JSON: \texttt{\{"job title","company name","work modality"","salary","summary"\}}. Usa \texttt{"DESCONOCIDO"} si falta. \\
    Responsabilidades & Extrae únicamente las responsabilidades. Responde array \verb|["resp1","resp2",...]|. \\
    Ubicación & Extrae únicamente la ubicación. Responde \texttt{\{"location":"..."\}}. Usa \texttt{"DESCONOCIDO"} si falta. \\
    Educación & Extrae únicamente los requisitos educativos. Responde \texttt{\{"education":"..."\}}. Usa \texttt{"DESCONOCIDO"} si falta. \\
    Experiencia & Extrae únicamente los requisitos de experiencia. Responde \texttt{\{"experience":"..."\}}. Usa \texttt{"DESCONOCIDO"} si falta. \\
    Habilidades técnicas & Extrae únicamente las habilidades técnicas. Responde array \texttt{["hab1","hab2",...]}. \\
    Habilidades blandas & Extrae únicamente las habilidades blandas. Responde array \texttt{["hab1","hab2",...]}. \\
    Certificaciones & Extrae únicamente las certificaciones requeridas. Responde array \texttt{["cert1","cert2",...]}. \\
    Idiomas & Extrae únicamente los requisitos de idiomas. Responde \texttt{\{"idioma":"nivel",...\}}. \\
    Beneficios & Extrae únicamente los beneficios. Responde array \texttt{["ben1","ben2",...]}. \\
    Inferencia soft skills & Infiere solo las \emph{soft skills} críticas desde las responsabilidades (máx. 3--5). Devuelve array \texttt{["soft1","soft2",...]}. \\
    Detección de idioma & Detecta el idioma principal del texto. Devuelve \texttt{\{"idioma":"Intermedio"\}}. \\
    \bottomrule
  \end{tabular}
\end{table}

\section*{Comparadores (\texttt{algoritmo\_recomendacion/comparators})}
\begin{table}[H]
  \centering
  \caption{Prompts de comparadores por aspecto}
  \label{tab:prompts-comp}
  \footnotesize
  \begin{tabular}{@{}lp{11cm}@{}}
    \toprule
    \textbf{Comparación} & \textbf{Prompt (resumen truncado)} \\
    \midrule
    Habilidades técnicas & Eres un experto en evaluar compatibilidad de habilidades técnicas. Responde solo JSON con \texttt{\{"score":0--1,"reason":"..."\}}. Compara el CV con las habilidades requeridas. \\
    Experiencia & Eres un experto en evaluar experiencia laboral. Responde solo JSON \texttt{\{"score":0--1,"reason":"..."\}} considerando años, roles y tecnologías. \\
    Responsabilidades & Eres un experto en compatibilidad entre responsabilidades y experiencia. Responde solo JSON \texttt{\{"score":0--1,"reason":"..."\}}. \\
    Educación & Eres un experto en compatibilidad educativa. Responde solo JSON \texttt{\{"score":0--1,"reason":"..."\}}. \\
    Idiomas & Eres un experto en niveles de idioma. Responde solo JSON \texttt{\{"compatible":true,"score":0--1,"reason":"..."\}}. \\
    Ubicación & Eres un experto en ubicaciones geográficas. Responde solo JSON \texttt{\{"score":0--1,"reason":"..."\}}. La misma ciudad o país equivale a 1.0; países lejanos a 0.0. \\
    Habilidades blandas & Eres un experto en evaluar habilidades blandas. Responde solo JSON \texttt{\{"score":0--1,"reason":"..."\}}. \\
    Certificaciones & Eres un experto en evaluar certificaciones técnicas. Responde solo JSON \texttt{\{"score":0--1,"reason":"..."\}} comparando su relevancia frente a los requisitos o habilidades. \\
    \bottomrule
  \end{tabular}
\end{table}
